# include <stdio.h>
int main()
{

  //Simple interest program 
  float principle, rate, time, interest;
  
  printf("Enter the  principle:");
  scanf("%f", &principle);

  printf("Enter the rate:");
  scanf(("%f", &rate);

  printf("Enter the time:");
  scanf("%f", &time);

  interest=(principle*rate*time)/100;
  printf("The simple interest is %f", interest); 
  return 0;
}
